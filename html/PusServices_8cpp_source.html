118CCSDS::ResultBool PusC::deserialize(const std::vector<std::uint8_t> &data) {
119 RET_IF_ERR_MSG(data.size() <= m_size, CCSDS::ErrorCode::INVALID_SECONDARY_HEADER_DATA,
120 "PUS-C header not correct size (size <= 6 bytes)");
121 m_version = data[0] & 0x7;
122 m_serviceType = data[1];
123 m_serviceSubType = data[2];
124 m_sourceID = data[3];
125 m_dataLength = data[data.size()-2] << 8 | data[data.size()-1];
126 if (data.size() > 6) {
127 m_timeCode.assign(data.begin() + 4, data.end()-2);
128 }
129 return true;
130}
