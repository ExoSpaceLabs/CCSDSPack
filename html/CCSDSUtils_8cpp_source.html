12uint16_t crc16(
13 const std::vector<std::uint8_t> &data, const std::uint16_t polynomial, const std::uint16_t initialValue,
14 const std::uint16_t finalXorValue) {
15 std::uint16_t crc = initialValue;
16
17 for (const auto &byte: data) {
18 crc ^= static_cast<std::uint16_t>(byte) << 8; // Align byte with MSB of 16-bit CRC
19 for (std::int32_t i = 0; i < 8; ++i) {
20 // Process each bit
21 if (crc & 0x8000) {
22 // Check if MSB is set
23 crc = (crc << 1) ^ polynomial; // Shift and XOR with polynomial
24 } else {
25 crc = crc << 1; // Shift only
26 }
27 }
28 }
29
30 return crc ^ finalXorValue; // Apply final XOR
31}

38std::string getBinaryString(const std::uint32_t value, const std::int32_t bits) {
39 std::string binaryString;
40 // Calculate the minimum number of bits required to represent in groups of 4
41 const std::int32_t paddedBits = ((bits + 3) / 4) * 4; // Round up to the nearest multiple of 4
42
43 for (std::int32_t i = paddedBits - 1; i >= 0; --i) {
44 binaryString += ((value >> i) & 1) ? '1' : '0';
45
46 // Add a space after every 4 bits, except at the end
47 if (i % 4 == 0 && i != 0) {
48 binaryString += ' ';
49 }
50 }
51 return binaryString;
52}

214CCSDS::ResultBuffer readBinaryFile(const std::string& filename) {
215 RET_IF_ERR_MSG(filename.empty(),CCSDS::ErrorCode::FILE_READ_ERROR, "No filename provided");
216
217 std::ifstream in(filename, std::ios::binary | std::ios::ate);
218 RET_IF_ERR_MSG(!in,CCSDS::ErrorCode::FILE_READ_ERROR, "Failed to open file for reading");
219
220 // Get the file size using the 'ate' flag (seeks to the end automatically)
221 const std::streamsize size = in.tellg();
222 in.seekg(0, std::ios::beg);
223
224 // Read the entire file content into the vector
225 std::vector<std::uint8_t> data(size);
226 in.read(reinterpret_cast<char*>(data.data()), size);
227
228 RET_IF_ERR_MSG(!in,CCSDS::ErrorCode::FILE_READ_ERROR, "Failed to read the entire file");
229 return data;
230}
