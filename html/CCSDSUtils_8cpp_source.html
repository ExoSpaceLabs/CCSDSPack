10std::string getBinaryString(const uint32_t value, const int bits) {
11 std::string binaryString;
12 // Calculate the minimum number of bits required to represent in groups of 4
13 const int paddedBits = ((bits + 3) / 4) * 4; // Round up to the nearest multiple of 4
14
15 for (int i = paddedBits - 1; i >= 0; --i) {
16 binaryString += ((value >> i) & 1) ? '1' : '0';
17
18 // Add a space after every 4 bits, except at the end
19 if (i % 4 == 0 && i != 0) {
20 binaryString += ' ';
21 }
22 }
23 return binaryString;
24}

111uint16_t crc16(
112 const std::vector<uint8_t> &data, const uint16_t polynomial, const uint16_t initialValue,
113 const uint16_t finalXorValue) {
114 uint16_t crc = initialValue;
115
116 for (const auto &byte: data) {
117 crc ^= static_cast<uint16_t>(byte) << 8; // Align byte with MSB of 16-bit CRC
118 for (int i = 0; i < 8; ++i) {
119 // Process each bit
120 if (crc & 0x8000) {
121 // Check if MSB is set
122 crc = (crc << 1) ^ polynomial; // Shift and XOR with polynomial
123 } else {
124 crc = crc << 1; // Shift only
125 }
126 }
127 }
128
129 return crc ^ finalXorValue; // Apply final XOR
130}

185CCSDS::ResultBool Config::load(const std::string &filename) {
186 std::ifstream file(filename);
187 RET_IF_ERR_MSG(!file.is_open(),CCSDS::ErrorCode::CONFIG_FILE_ERROR, "Failed to open configuration file");
188
189 std::string line;
190 while (std::getline(file, line)) {
191 if (line.empty() || line.front() == '#') continue;
192
193 auto [key, type, valueStr] = parseLine(line);
194 RET_IF_ERR_MSG(key.empty() || type.empty(),CCSDS::ErrorCode::CONFIG_FILE_ERROR, "Failed to parse configuration file");
195 ConfigValue value;
196 if (type == "string") {
197 value = valueStr;
198 } else if (type == "int") {
199 value = std::stoi(valueStr);
200 } else if (type == "float") {
201 value = std::stof(valueStr);
202 } else if (type == "bool") {
203 value = (valueStr == "true" || valueStr == "1");
204 } else if (type == "bytes") {
205 ASSIGN_CP( value, parseBytes(valueStr) );
206 } else {
207 return CCSDS::Error{CCSDS::ErrorCode::CONFIG_FILE_ERROR, " unknown type: " + type};
208 }
209
210 values[key] = value;
211 }
212
213 return true;
214}

216std::tuple<std::string, std::string, std::string> Config::parseLine(const std::string& line) {
217 auto colonPos = line.find(':');
218 auto equalPos = line.find('=');
219
220 if (colonPos == std::string::npos || equalPos == std::string::npos || equalPos < colonPos)
221 return {"", "", ""};
222
223 std::string key = line.substr(0, colonPos);
224 std::string type = line.substr(colonPos + 1, equalPos - colonPos - 1);
225 std::string value = line.substr(equalPos + 1);
226
227 if (!value.empty() && value.front() == '"' && value.back() == '"')
228 value = value.substr(1, value.size() - 2);
229
230 return {key, type, value};
231}

233CCSDS::ResultBuffer Config::parseBytes(const std::string &valueStr) {
234 std::vector<uint8_t> result;
235 RET_IF_ERR_MSG(valueStr.front() != '[' || valueStr.back() != ']', CCSDS::ErrorCode::CONFIG_FILE_ERROR, "Config: Invalid buffer formatting []");
236
237 std::string inner = valueStr.substr(1, valueStr.size() - 2);
238 std::stringstream ss(inner);
239 std::string token;
240
241 while (std::getline(ss, token, ',')) {
242 // Trim whitespace
243 token.erase(std::remove_if(token.begin(), token.end(), ::isspace), token.end());
244
245 try {
246 int base = 10;
247 if (token.size() > 2 && token.substr(0, 2) == "0x") {
248 token = token.substr(2);
249 base = 16;
250 } else if (std::all_of(token.begin(), token.end(), ::isxdigit) && token.size() <= 2) {
251 base = 16; // Assume short tokens with hex chars are meant to be hex
252 }
253 result.push_back(static_cast<uint8_t>(std::stoi(token, nullptr, base)));
254 } catch (...) {
255 return CCSDS::Error{CCSDS::ErrorCode::CONFIG_FILE_ERROR, "Invalid byte value: " + token};
256 }
257 }
258 return result;
259}
