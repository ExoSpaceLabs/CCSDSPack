297CCSDS::ResultBool parseArguments(const int argc, char *argv[],
298 std::unordered_map<std::string, std::string> &allowedMap,
299 std::unordered_map<std::string, std::string> &outArgs)
300{
301 const std::set<std::string> booleanArgs{"verbose", "help"};
302 std::set<std::string> allowedKeys;
303 std::set<std::string> allowedShortKeys;
304 for (const auto& [k, v] : allowedMap) {
305 allowedShortKeys.insert(k);
306 allowedKeys.insert(v);
307 }
308 for (int i = 1; i < argc; ++i) {
309 std::string current = argv[i];
310
311 // Check if this is a key
312 if (current.rfind("--", 0) == 0 || current.rfind('-', 0) == 0) {
313 std::string key;
314 if (current.rfind("--", 0) == 0){
315 key = current.substr(2);
316 }else {
317 std::string sKey = current.substr(1);
318 RET_IF_ERR_MSG(allowedShortKeys.find(sKey) == allowedShortKeys.end(), static_cast<CCSDS::ErrorCode>(ARG_PARSE_ERROR), "Unknown argument: -" + sKey);
319 key = allowedMap.at(sKey);
320 }
321 RET_IF_ERR_MSG(allowedKeys.find(key) == allowedKeys.end(), static_cast<CCSDS::ErrorCode>(ARG_PARSE_ERROR), "Unknown argument: --" + key);
322
323 if (booleanArgs.find(key) != booleanArgs.end()) {
324 outArgs[key] = "true";
325 }else {
326 RET_IF_ERR_MSG(i + 1 >= argc, static_cast<CCSDS::ErrorCode>(ARG_PARSE_ERROR), "Missing value for argument: --" + key);
327 const std::string value = argv[++i];
328 outArgs[key] = value;
329 }
330 }else {
331 return CCSDS::Error{ static_cast<CCSDS::ErrorCode>(ARG_PARSE_ERROR), "Unknown argument:" + std::string(argv[i]) };
332 }
333 }
334 return true;
335}

338 // Get current timestamp with high precision

340 const auto duration = now.time_since_epoch();
