138CCSDS::ResultBool parseArguments(const int argc, char *argv[],
139 std::unordered_map<std::string, std::string> &allowedMap,
140 std::unordered_map<std::string, std::string> &outArgs)
141{
142 const std::set<std::string> booleanArgs{"verbose", "help"};
143 std::set<std::string> allowedKeys;
144 std::set<std::string> allowedShortKeys;
145 for (const auto& [k, v] : allowedMap) {
146 allowedShortKeys.insert(k);
147 allowedKeys.insert(v);
148 }
149 for (int i = 1; i < argc; ++i) {
150 std::string current = argv[i];
151
152 // Check if this is a key
153 if (current.rfind("--", 0) == 0 || current.rfind('-', 0) == 0) {
154 std::string key;
155 if (current.rfind("--", 0) == 0){
156 key = current.substr(2);
157 }else {
158 std::string sKey = current.substr(1);
159 RET_IF_ERR_MSG(allowedShortKeys.find(sKey) == allowedShortKeys.end(), static_cast<CCSDS::ErrorCode>(ARG_PARSE_ERROR), "Unknown argument: -" + sKey);
160 key = allowedMap.at(sKey);
161 }
162 RET_IF_ERR_MSG(allowedKeys.find(key) == allowedKeys.end(), static_cast<CCSDS::ErrorCode>(ARG_PARSE_ERROR), "Unknown argument: --" + key);
163
164 if (booleanArgs.find(key) != booleanArgs.end()) {
165 outArgs[key] = "true";
166 }else {
167 RET_IF_ERR_MSG(i + 1 >= argc, static_cast<CCSDS::ErrorCode>(ARG_PARSE_ERROR), "Missing value for argument: --" + key);
168 const std::string value = argv[++i];
169 outArgs[key] = value;
170 }
171 }else {
172 return CCSDS::Error{ static_cast<CCSDS::ErrorCode>(ARG_PARSE_ERROR), "Unknown argument:" + std::string(argv[i]) };
173 }
174 }
175 return true;
176}
