7void CCSDS::Packet::update() {
8 if (!m_updateStatus && m_enableUpdatePacket) {
9 const auto dataField = m_dataField.serialize();
10 const auto dataFiledSize = static_cast<uint16_t>(dataField.size());
11 const auto dataFieldHeaderFlag(m_dataField.getDataFieldHeaderFlag());
12 m_primaryHeader.setDataLength(dataFiledSize);
13 m_primaryHeader.setDataFieldHeaderFlag(dataFieldHeaderFlag);
14 // todo this part needs to be moved out of conditional updating
15 if (m_primaryHeader.getSequenceFlags() == UNSEGMENTED) {
16 m_primaryHeader.setSequenceCount(0);
17 } else {
18 m_primaryHeader.setSequenceCount(m_sequenceCounter);
19 }
20 m_CRC16 = crc16(dataField, m_CRC16Config.polynomial, m_CRC16Config.initialValue, m_CRC16Config.finalXorValue);
21 m_updateStatus = true;
22 }

109std::vector<uint8_t> CCSDS::Packet::getCRCVectorBytes() {

112 crc[0] = (crcVar >> 8) & 0xFF; // MSB (Most Significant Byte)

151std::vector<uint8_t> CCSDS::Packet::serialize() {
152 auto header = getPrimaryHeaderBytes();
153 auto dataField = m_dataField.serialize();
154 const auto crc = getCRCVectorBytes();
155
156 std::vector<uint8_t> packet;
157 packet.reserve(header.size() + dataField.size() + crc.size());
158 packet.insert(packet.end(), header.begin(), header.end());
159 if (!getFullDataFieldBytes().empty()) {
160 packet.insert(packet.end(), dataField.begin(), dataField.end());
161 }
162 packet.insert(packet.end(), crc.begin(), crc.end());
163
164 return packet;
165}

179CCSDS::ResultBool CCSDS::Packet::deserialize(const std::vector<uint8_t> &data, const std::string &headerType, int headerSize) {
180 RET_IF_ERR_MSG(data.size() <= 8, ErrorCode::INVALID_DATA,
181 "Cannot Deserialize Packet, Invalid Data provided data size must be at least 8 bytes");
182 RET_IF_ERR_MSG(headerType == "BufferHeader", ErrorCode::INVALID_SECONDARY_HEADER_DATA,
183 "Cannot Deserialize Packet, BufferHeader is not of defined size");
184 RET_IF_ERR_MSG(!m_dataField.getDataFieldHeaderFactory().typeIsRegistered(headerType),
185 ErrorCode::INVALID_SECONDARY_HEADER_DATA,
186 "Cannot Deserialize Packet, Unregistered Secondary header: " + headerType);
187 uint16_t headerDataSizeBytes{0};
188 const auto secondaryHeader = m_dataField.getDataFieldHeaderFactory().create(headerType);
189 if (headerType == "PusC" && headerSize > 0) {
190 headerDataSizeBytes = headerSize;
191 }else {
192 headerDataSizeBytes = secondaryHeader->getSize();
193 }
194
195 std::vector<uint8_t> dataFieldHeaderVector;
196 std::copy_n(data.begin() + 6, headerDataSizeBytes, std::back_inserter(dataFieldHeaderVector));
197 FORWARD_RESULT(secondaryHeader->deserialize(dataFieldHeaderVector ));
198 setDataFieldHeader(secondaryHeader);
199
200 if (data.size() > (8 + headerDataSizeBytes)) {
201 std::vector<uint8_t> dataFieldVector;
202
203 if (data.size() > (9 + headerDataSizeBytes)) {
204 std::copy(data.begin() + 6 + headerDataSizeBytes, data.end(), std::back_inserter(dataFieldVector));
205 }
206 FORWARD_RESULT(deserialize({data[0], data[1], data[2], data[3], data[4], data[5]}, dataFieldVector));
207 }
208
209 return true;
210}

212CCSDS::ResultBool CCSDS::Packet::deserialize(const std::vector<uint8_t> &data, const uint16_t headerDataSizeBytes) {
213 RET_IF_ERR_MSG(data.size() <= (8 + headerDataSizeBytes), ErrorCode::INVALID_DATA,

219 FORWARD_RESULT(m_dataField.setDataFieldHeader(secondaryHeader));

227CCSDS::ResultBool CCSDS::Packet::deserialize(const std::vector<uint8_t> &headerData, const std::vector<uint8_t> &data) {
228 RET_IF_ERR_MSG(headerData.size() != 6, ErrorCode::INVALID_HEADER_DATA,
229 "Cannot Deserialize Packet, Invalid Header Data provided.");
230 FORWARD_RESULT(m_primaryHeader.deserialize( headerData ));
231 m_sequenceCounter = m_primaryHeader.getSequenceCount();
232
233 RET_IF_ERR_MSG(data.size() < 2, ErrorCode::INVALID_DATA,
234 "Cannot Deserialize Packet, Invalid Data provided, at least CRC is required.");
235
236 std::vector<uint8_t> dataCopy;
237 m_CRC16 = (data[data.size() - 2] << 8) + data.back();
238
239 if (data.size() == 2) return true; // returns since no application data is to be written.
240
241 std::copy(data.begin(), data.end() - 2, std::back_inserter(dataCopy));
242 FORWARD_RESULT(m_dataField.setApplicationData(dataCopy));
243
244 return true;;
245}
